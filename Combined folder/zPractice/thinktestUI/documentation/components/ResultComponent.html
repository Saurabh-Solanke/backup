import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';
import { Router, RouterOutlet } from '@angular/router';
import Swal from 'sweetalert2';
import { HttpClientModule } from '@angular/common/http';
import { HostListener } from '@angular/core';
import { Chart, registerables } from 'chart.js';

import * as Papa from 'papaparse';
import { BrowserStorageService } from '../../service/browser-storage.service';

import { LoggedInFooterComponent } from '../../shared/components/logged-in-footer/logged-in-footer.component';
import { TestResultService } from '../../service/test-result-service.service';

Chart.register(...registerables);

@Component({
  selector: 'app-result',
  standalone: true,
  imports: [CommonModule, RouterOutlet, HttpClientModule, LoggedInFooterComponent],
  templateUrl: './result.component.html',
  styleUrls: ['./result.component.css']
})
export class ResultComponent {
  questions: any[] = [];
  userResponses: any[] = [];
  attempted: number = 0;
  correct: number = 0;
  incorrect: number = 0;
  percentage: number = 0;
  timeTaken: number = 0;
  timeTakenStr: string = '';
  chart: any;
  
  name: string | null;

  constructor(private router: Router, private sessionStorageService: BrowserStorageService, private testResultService: TestResultService) {
    this.name = this.sessionStorageService.getItem("loggedInUser");
    const navigation = this.router.getCurrentNavigation();
    const state = navigation?.extras.state as { userResponses: any[], questions: any[], timeTaken: number };
    if (state) {
      this.userResponses = state.userResponses;
      this.questions = state.questions;
      this.timeTaken = state.timeTaken;
      this.timeTakenStr = this.formatTime(this.timeTaken);
    } else {
      this.router.navigate(['/']); // Redirect to home if no state is found
    }

    const userId = this.sessionStorageService.getItem('loggedInUserId'); 
    this.testResultService.getUser(userId!).subscribe(user => {
      const testResult = {
        date: new Date(),
        score: this.correct,
        totalQuestions: this.questions.length,
        timeTaken: this.timeTaken
      };

      user.testHistory = user.testHistory || [];
      user.testHistory.push(testResult);

      this.testResultService.updateUser(userId!, user).subscribe(() => {
        Swal.fire('Success', 'Test result saved successfully!', 'success').then(() => {
          // this.router.navigate(['/testHistory']);
        });
      });
    });

  }

  ngOnInit(): void {
    this.calculateResult();
    this.createChart();
  }

  calculateResult() {
    this.attempted = this.userResponses.filter(response => response !== null).length;
    this.correct = this.questions.filter((q, i) => q.answer === this.userResponses[i]).length;
    this.incorrect = this.attempted - this.correct;
    this.percentage = (this.correct / this.questions.length) * 100;
  }

  createChart() {
    const unattempted = this.questions.length - this.attempted;
    const data = [this.correct, this.incorrect, unattempted];

    this.chart = new Chart('resultChart', {
      type: 'pie',
      data: {
        labels: ['Correct', 'Incorrect', 'Unattempted'],
        datasets: [{
          data: data,
          backgroundColor: ['#4caf50', '#f44336', '#ff9800'],
        }]
      },
      options: {
        responsive: true,
        plugins: {
          legend: {
            position: 'top',
          },
          title: {
            display: true,
            text: 'Exam Results'
          }
        }
      }
    });
  }

  formatTime(seconds: number): string {
    const minutes = Math.floor(seconds / 60);
    const remainingSeconds = seconds % 60;
    return `${minutes}m ${remainingSeconds}s`;
  }

  exit() {
    Swal.fire('Thank You', 'You have completed the exam!', 'success').then(() => {
      sessionStorage.removeItem('examStarted');
      sessionStorage.removeItem('examFinished');
      this.router.navigate(['/home']);
    });
  }

  @HostListener('window:popstate', ['$event'])
  onPopState($event: any): void {
    Swal.fire({
      title: 'Navigation Alert',
      text: 'You cannot go back to the test. You will be redirected to the home page.',
      icon: 'warning',
      confirmButtonText: 'OK'
    }).then(() => {
      this.router.navigate(['/']);
    });
  }

  downloadCsvResponseSheet() {
    const data = this.questions.map((question, index) => {
      return {
        Question: question.question,
        Correct_Answer: question.options[question.answer],
        Your_Response: this.userResponses[index] !== null ? question.options[this.userResponses[index]] : 'Unattempted'
      };
    });

    const csv = Papa.unparse(data);
    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'Response_Sheet.csv';
    a.click();
    URL.revokeObjectURL(url);
  }

  printPage() {
    window.print();
  }
  
}
