import { Component } from '@angular/core';
import { FormBuilder, FormGroup, ReactiveFormsModule, Validators } from '@angular/forms';
import { Router, RouterLink, RouterOutlet } from '@angular/router';
import { AuthService } from '../../service/auth.service';
import { CommonModule } from '@angular/common';
import { SignupComponent } from '../signup/signup.component';
import {  HttpClientModule } from '@angular/common/http';
import Swal from 'sweetalert2';
import { BrowserStorageService } from '../../service/browser-storage.service';


@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  standalone: true,
  imports:[ReactiveFormsModule,CommonModule,SignupComponent,RouterOutlet,RouterLink,HttpClientModule],
  providers: [AuthService],
  styleUrls: ['./login.component.css']
})
export class LoginComponent {
  loginForm: FormGroup;
  errorMessage: string = '';

  constructor(
    private fb: FormBuilder,
    private authService: AuthService,
    private router: Router,
    private sessionStorageService : BrowserStorageService
  ) {
    this.loginForm = this.fb.group({
      email: ['', [Validators.required, Validators.email]],
      password: ['', Validators.required]
    });
  }

  onSubmit() {
    if (this.loginForm.valid) {
      this.authService.login(this.loginForm.value).subscribe(
        (response: any) => {
          if (response.password === this.loginForm.value.password) {
            Swal.fire('Success', 'Login successful!', 'success');
            // console.log(response);
           this.sessionStorageService.setItem('loggedInUser', response.name); 
           this.sessionStorageService.setItem('loggedInUserId', response.id); 
           
            this.router.navigate(['/home'], { state: { username: response.name } });
          } else {
            Swal.fire('Error', 'Invalid email or password', 'error');
          }
        },
        error => {
          Swal.fire('Error', 'An error occurred. Please try again.', 'error');
        }
      );
    } else {
      Swal.fire('Error', 'Please fill in all the fields correctly.', 'error');
    }
  }

  // redirectToSignup() {
  //   this.router.navigate(['/regsiter']);
  // }
}
