import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { BrowserStorageService } from '../../service/browser-storage.service';
import { TestResultService } from '../../service/test-result-service.service';
import { TimeAgoPipe } from '../../shared/pipes/time-ago.pipe';
import { LoggedInFooterComponent } from '../../shared/components/logged-in-footer/logged-in-footer.component';
import { FormatTimePipe } from '../../shared/pipes/format-time.pipe';
import { RouterModule } from '@angular/router';

@Component({
  selector: 'app-test-history',
  standalone: true,
  imports: [CommonModule, LoggedInFooterComponent, TimeAgoPipe, FormatTimePipe, RouterModule],
  templateUrl: './test-history.component.html',
  styleUrls: ['./test-history.component.css'],
})
export class TestHistoryComponent implements OnInit {
  name: string | null;
  testHistory: any[] = [];

  constructor(private sessionStorageService: BrowserStorageService, private testResultService: TestResultService) {
    this.name = this.sessionStorageService.getItem('loggedInUser');
  }

  ngOnInit(): void {
    this.loadTestHistory();
  }

  loadTestHistory() {
    const userId = this.sessionStorageService.getItem('loggedInUserId'); 
    this.testResultService.getUser(userId!).subscribe(user => {
      this.testHistory = user.testHistory || [];
      this.testHistory.reverse();
    });
  }
}
