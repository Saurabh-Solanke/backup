import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule } from '@angular/forms';
import { HttpClientModule } from '@angular/common/http';
import { Router, RouterOutlet } from '@angular/router';
import Swal from 'sweetalert2';
import { QuestionService } from '../../service/question.service';
import { HostListener } from '@angular/core';
import { BrowserStorageService } from '../../service/browser-storage.service';
import { Question } from '../../shared/interfaces/Question';

import { LoggedInFooterComponent } from '../../shared/components/logged-in-footer/logged-in-footer.component';

@Component({
  selector: 'app-exam',
  standalone: true,
  imports: [CommonModule, FormsModule, RouterOutlet, HttpClientModule, LoggedInFooterComponent],
  providers: [QuestionService],
  templateUrl: './exam.component.html',
  styleUrls: ['./exam.component.css']
})
export class ExamComponent {
  questions: Question[] = [];
  userResponses: (number | null)[] = [];
  currentQuestionIndex = 0;
  timer: number = 30 * 60; // 30 minutes in seconds
  timerInterval: any;
  minutes: number = 0;
  seconds: number = 0;
  attempted: number = 0;
  remaining: number = 0;
  
  name: string | null;

  constructor(private questionService: QuestionService, private router: Router, private sessionStorageService: BrowserStorageService) {
    this.preventBackButton();
    this.name = this.sessionStorageService.getItem("loggedInUser");
  }

  ngOnInit(): void {
    this.fetchQuestions(); // Fetch questions when component initializes
    this.preventBackButton();
  }

  fetchQuestions() {
    this.questionService.getQuestions().subscribe(data => {
      this.questions = data;
      this.userResponses = new Array(data.length).fill(null);
      this.updateStatus();
      this.startTimer();
    }, error => {
      console.error('Error fetching questions:', error);
      Swal.fire('Error', 'Failed to load questions. Please try again later.', 'error');
    });
  }

  startTimer() {
    this.timerInterval = setInterval(() => {
      this.timer--;
      if (this.timer === 600) {
        Swal.fire('Alert', '10 minutes remaining!', 'warning');
      }
      if (this.timer <= 0) {
        clearInterval(this.timerInterval);
        this.submitExam();
      }
      this.minutes = Math.floor(this.timer / 60);
      this.seconds = this.timer % 60;
    }, 1000);
  }

  selectOption(index: number) {
    this.userResponses[this.currentQuestionIndex] = index; // Store selected option index
    this.updateStatus();
  }

  updateStatus() {
    this.attempted = this.userResponses.filter(response => response !== null).length;
    this.remaining = this.questions.length - this.attempted;
  }

  clearResponse() {
    this.userResponses[this.currentQuestionIndex] = null; // Clear user's response for current question
    this.updateStatus(); // Update attempted and remaining counts
  }

  previousQuestion() {
    if (this.currentQuestionIndex > 0) {
      this.currentQuestionIndex--;
    }
  }

  nextQuestion() {
    if (this.currentQuestionIndex < this.questions.length - 1) {
      this.currentQuestionIndex++;
    }
  }

  jumpToQuestion(index: number) {
    this.currentQuestionIndex = index;
  }

  submitExam() {

    // const answeredCount = this.userResponses.filter(response => response !== null).length;
    // const minRequired = Math.ceil(this.questions.length * 0.8);
  
    // if (answeredCount < minRequired) {
    //   Swal.fire('Error', `You must answer at least 80% of the questions before submitting.`, 'error');
    //   return;
    // }
  
    Swal.fire({
      title: 'Submit Exam?',
      text: "Are you sure you want to submit?",
      icon: 'warning',
      showCancelButton: true,
      confirmButtonText: 'Yes, submit!',
      cancelButtonText: 'No, cancel'
    }).then((result) => {
      if (result.isConfirmed) {
        clearInterval(this.timerInterval);
        sessionStorage.setItem('examFinished', 'true');
        this.router.navigate(['/result'], { state: { userResponses: this.userResponses, questions: this.questions, timeTaken: 30 * 60 - this.timer } });
      }
    });
  }

  preventBackButton() {
    window.history.pushState(null, '', location.href);
    window.addEventListener('popstate', () => {
      window.history.pushState(null, '', location.href);
      Swal.fire({
        title: 'Warning',
        text: 'Navigating away will end your exam. Please submit the test first.',
        icon: 'warning',
        confirmButtonText: 'OK'
      });
    });
  }

  @HostListener('window:beforeunload', ['$event'])
  unloadNotification($event: any): void {
    $event.preventDefault();
    $event.returnValue = 'You have unsaved changes!'; // This line is required for the browser to show the native alert
  }

  @HostListener('window:popstate', ['$event'])
  onPopState($event: any): void {
    Swal.fire({
      title: 'Warning',
      text: 'Navigating away will end your exam. Please submit the test first.',
      icon: 'warning',
      confirmButtonText: 'OK'
    }).then(() => {
      window.history.pushState(null, '', location.href);
    });
  }

}
