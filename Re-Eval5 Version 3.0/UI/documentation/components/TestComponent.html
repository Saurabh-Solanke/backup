import { Component, OnDestroy, OnInit } from '@angular/core';
import { IExamLive } from '../../../core/interfaces/exam';
import { IQuestion } from '../../../core/interfaces/exam';
import { ApiService } from '../../../core/services/api.service';
import { Router } from '@angular/router';
import { ToastrService } from 'ngx-toastr';
import Swal from 'sweetalert2';
import { Subscription } from 'rxjs';

@Component({
  selector: 'app-test',
  templateUrl: './test.component.html',
  styleUrl: './test.component.css'
})
export class TestComponent implements OnInit, OnDestroy {
  exam!: IExamLive;
  questions: IQuestion[] = [];
  currentQuestionIndex = 0;
  userResponses: any[] = [];
  reviewedQuestions: boolean[] = [];
  minutes = 0;
  seconds = 0;
  timerInterval: any;
  timer = 30 * 60; // 30 minutes in seconds
  examId!: number;
  attempted = 0;
  remaining = 0;
  subscriber$ = new Subscription();

  constructor(
    private examService: ApiService,
    private router: Router,
    private toastr: ToastrService
  ) {}

  ngOnInit(): void {
    this.getExamId();
    this.fetchExamDetails();
  }

  ngOnDestroy(): void {
    if (this.timerInterval) {
      clearInterval(this.timerInterval);
    }
    this.subscriber$.unsubscribe();
  }

  getExamId(): void {
    const storedExamId = sessionStorage.getItem('examId');
    if (storedExamId) {
      this.examId = Number(storedExamId);
    } else {
      this.toastr.error('No exam selected!', 'Error', { progressBar: true });
      this.router.navigate(['/user/choose-exam']);
    }
  }

  fetchExamDetails(): void {
    this.subscriber$.add(
      this.examService.examGetExamDetailsByExamId(this.examId).subscribe({
        next: (data: IExamLive) => {
          this.exam = data;
          this.questions = this.extractQuestions(data);
          this.initializeUserResponses();
          this.startTimer();
        },
        error: (err) => {
          console.error('Error fetching exam details:', err);
          this.toastr.error('Error fetching exam details', 'Error', {
            progressBar: true,
          });
        }
      })
    );
  }

  extractQuestions(exam: IExamLive): IQuestion[] {
    return exam.sections.flatMap((section) => section.questions);
  }

  initializeUserResponses(): void {
    this.userResponses = new Array(this.questions.length).fill(null);
    this.reviewedQuestions = new Array(this.questions.length).fill(false);
    this.calculateAttemptedAndRemaining();
  }

  startTimer(): void {
    this.timerInterval = setInterval(() => {
      this.timer--;
      if (this.timer === 600) {
        this.toastr.warning('10 minutes remaining!', 'Warning', {
          progressBar: true,
        });
      }
      if (this.timer <= 0) {
        clearInterval(this.timerInterval);
        // this.timeUpAlert();
      }
      this.minutes = Math.floor(this.timer / 60);
      this.seconds = this.timer % 60;
    }, 1000);
  }

  selectOption(optionIndex: number): void {
    this.userResponses[this.currentQuestionIndex] = optionIndex;
    this.calculateAttemptedAndRemaining();
  }

  updateMultipleChoiceAnswer(questionIndex: number, optionIndex: number, event: any): void {
    if (!this.userResponses[questionIndex]) {
      this.userResponses[questionIndex] = [];
    }

    if (event.target.checked) {
      this.userResponses[questionIndex].push(optionIndex);
    } else {
      const optionPosition = this.userResponses[questionIndex].indexOf(optionIndex);
      if (optionPosition !== -1) {
        this.userResponses[questionIndex].splice(optionPosition, 1);
      }
    }

    this.calculateAttemptedAndRemaining();
  }

  calculateAttemptedAndRemaining(): void {
    this.attempted = this.userResponses.filter(response => response !== null && (Array.isArray(response) ? response.length > 0 : true)).length;
    this.remaining = this.questions.length - this.attempted;
  }

  clearResponse(): void {
    this.userResponses[this.currentQuestionIndex] = null;
    this.calculateAttemptedAndRemaining();
  }

  toggleReviewStatus(): void {
    this.reviewedQuestions[this.currentQuestionIndex] = !this.reviewedQuestions[this.currentQuestionIndex];
  }

  navigateQuestion(direction: 'previous' | 'next'): void {
    switch (direction) {
      case 'previous':
        if (this.currentQuestionIndex > 0) {
          this.currentQuestionIndex--;
        }
        break;
      case 'next':
        if (this.currentQuestionIndex < this.questions.length - 1) {
          this.currentQuestionIndex++;
        }
        break;
    }
  }

  jumpToQuestion(index: number): void {
    this.currentQuestionIndex = index;
  }

  submitExam(): void {
    const totalQuestions = this.questions.length;
    const attemptedQuestions = this.userResponses.filter(response => response !== null).length;
    const requiredAttempts = Math.ceil(totalQuestions * 0.8); // 80% of total questions

    if (attemptedQuestions < requiredAttempts) {
      Swal.fire({
        title: 'Insufficient Attempts',
        text: `You must attempt at least ${requiredAttempts} questions to submit the exam. Currently, you have attempted ${attemptedQuestions}.`,
        icon: 'warning',
        confirmButtonText: 'OK'
      });
    } else {
      Swal.fire({
        title: 'Submit Exam?',
        text: "Are you sure you want to submit?",
        icon: 'warning',
        showCancelButton: true,
        confirmButtonText: 'Yes, submit!',
        cancelButtonText: 'No, cancel'
      }).then((result) => {
        if (result.isConfirmed) {
          clearInterval(this.timerInterval);
          this.sendExamSubmission(); // Send the submission to the backend
        }
      });
    }
  }

  sendExamSubmission(): void {
    const responses: any[] = this.questions.map((question, index) => {
      if (this.userResponses[index] !== null) {
        return {
          questionId: question.questionID,
          selectedOptions: Array.isArray(this.userResponses[index])
            ? this.userResponses[index]
            : [this.userResponses[index]]
        };
      }
      return null;
    }).filter(response => response !== null);

    // Further submission logic here...
  }
}

                        'btn-success': userResponses[i] !== null && !reviewedQuestions[i],
                        'btn-warning': reviewedQuestions[i],
